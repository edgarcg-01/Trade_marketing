apps/view/src/app/core/http/auth.interceptor.ts

authInterceptor
Type : HttpInterceptorFn
Default value : (req, next) => { const authService = inject(AuthService); const router = inject(Router); // Skip interceptor if calling an open route like /auth/login if (req.url.includes('/auth/login')) { return next(req); } const token = authService.token(); let modifiedReq = req; if (token) { modifiedReq = req.clone({ setHeaders: { Authorization: `Bearer ${token}` } }); } return next(modifiedReq).pipe( catchError((error: HttpErrorResponse) => { // Si el backend dictamina un 401 Unauthorized, forzamos logout (la regla del JWT) if (error.status === 401) { authService.logout(); router.navigate(['/login']); } return throwError(() => error); }) ); }
